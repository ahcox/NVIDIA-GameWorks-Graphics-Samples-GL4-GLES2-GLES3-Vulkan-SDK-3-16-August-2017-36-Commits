00023     // Projection matrix creation (Vulkan standard)
00024     // From the frustum definition
00025     //
00026     //   return the projection matrix
00028     template<class T>
00029     matrix4<T>& frustumVk(matrix4<T> & M, const T l, const T r, const T b, const T t, const T n, const T f)
00030     {
00031         M.element(0, 0) = ((T)(2.0))*n / (r - l);
00032         M.element(1, 0) = 0.0;
00033         M.element(2, 0) = 0.0;
00034         M.element(3, 0) = 0.0;
00035 
00036         M.element(0, 1) = 0.0;
00037         M.element(1, 1) = ((T)(2.0))*n / (b - t);
00038         M.element(2, 1) = 0.0;
00039         M.element(3, 1) = 0.0;
00040 
00041         M.element(0, 2) = (r + l) / (r - l);
00042         M.element(1, 2) = (b + t) / (b - t);
00043         M.element(2, 2) = -f / (f - n);
00044         M.element(3, 2) = -1.0;
00045 
00046         M.element(0, 3) = 0.0;
00047         M.element(1, 3) = 0.0;
00048         M.element(2, 3) = -(f*n) / (f - n);
00049         M.element(3, 3) = 0.0;
00050 
00051         return M;
00052     }

00056     // From the fovy in radians, aspect ratio and near far definition
00057     //
00058     //   return the projection matrix
00060     template<class T>
00061     matrix4<T>& perspectiveVk(matrix4<T> & M, const T fovy, const T aspect, const T n, const T f)
00062     {
00063         T xmin, xmax, ymin, ymax;
00064 
00065         ymax = n * (T)tan(fovy * 0.5);
00066         ymin = -ymax;
00067 
00068         xmin = ymin * aspect;
